Tourism RAG 项目操作指南  (v2.0 多智能体版)

  启动 & 停止

  # 首次启动 / 全量启动
  docker compose --env-file .env up --build -d

  # 日常启动（不重建镜像）
  docker compose --env-file .env up -d

  # 停止所有服务（保留数据）
  docker compose down

  # 停止并清除所有数据卷（慎用！会清空数据库和向量库）
  docker compose down -v

  ---
  重建单个服务

  # 只重建前端（改了 Vue 代码后）
  docker compose --env-file .env up --build -d frontend

  # 只重建后端（改了 Java 代码后）
  docker compose --env-file .env up --build -d backend

  ---
  查看状态 & 日志

  # 查看所有容器状态
  docker compose ps

  # 查看后端日志（实时跟踪）
  docker logs -f tourism-backend

  # 查看前端日志
  docker logs -f tourism-frontend

  # 查看后端最近 100 行日志
  docker logs tourism-backend --tail=100

  # 查看所有服务日志
  docker compose logs -f

  ★ 多智能体系统 (v2.0)

  # 查看智能体注册状态（应返回 8 个智能体）
  curl http://localhost:8080/api/multi-agent/status | jq

  # 列出所有智能体
  curl http://localhost:8080/api/multi-agent/agents | jq

  # 非流式多智能体行程生成
  curl -X POST http://localhost:8080/api/multi-agent/itinerary \
    -H 'Authorization: Bearer <TOKEN>' \
    -H 'Content-Type: application/json' \
    -d '{
      "cityCode":"qingdao",
      "startDate":"2026-06-10",
      "endDate":"2026-06-12",
      "preferences":["food","photography"],
      "budget":"medium",
      "transportMode":"transit"
    }'

  # ★ SSE 流式多智能体行程生成（可实时看智能体活动）
  curl -N -X POST http://localhost:8080/api/multi-agent/itinerary/stream \
    -H 'Authorization: Bearer <TOKEN>' \
    -H 'Content-Type: application/json' \
    -d '{
      "cityCode":"qingdao",
      "startDate":"2026-06-10",
      "endDate":"2026-06-12",
      "preferences":["food","photography"],
      "budget":"medium",
      "transportMode":"transit"
    }'

  # 标准模式行程生成（原接口，仍然可用）
  curl -X POST http://localhost:8080/api/agent/itinerary \
    -H 'Authorization: Bearer <TOKEN>' \
    -H 'Content-Type: application/json' \
    -d '{
      "cityCode":"qingdao",
      "startDate":"2026-06-10",
      "endDate":"2026-06-12",
      "preferences":["food","photography"],
      "budget":"medium",
      "transportMode":"transit"
    }'

  ---
  前端使用多智能体模式

  1. 打开 http://localhost:3000/itinerary
  2. 侧边栏找到"引擎模式"
  3. 点击"🤖 多智能体"切换到多智能体模式
  4. 点击"启动多智能体"按钮
  5. 观察仪表盘实时展示 8 个智能体的协作过程

  知识库操作

  # 进入前端 → 顶栏点"知识库" → 上传文件
  # 支持 .md / .txt / .pdf 格式

  # 如果 Milvus 集合损坏，手动清除（慎用）
  docker run --rm --network tourism-rag_tourism-net \
    python:3.11-slim bash -c \
    "pip install pymilvus -q && python -c \
  from pymilvus import connections, utility
  connections.connect(host='milvus', port='19530')
  utility.drop_collection('tourism_knowledge_v2')
  print('已清除')
  \"

  docker exec -it tourism-mysql mysql -u tourism -ptourism_pass_2024 tourism_rag

  docker exec tourism-mysql mysqldump -u root -ptourism_root_2024 tourism_rag > \
  backup.sql

  docker exec -i tourism-mysql mysql -u root -ptourism_root_2024 tourism_rag < \
  backup.sql

  切换模型

  编辑 backend/src/main/resources/application.yml，修改 model-name，然后重建后端：

  chat-model:
    model-name: qwen-turbo   # 可选：qwen-turbo / qwen-plus / qwen-max

  docker compose --env-file .env up --build -d backend

  ---
  多智能体配置

  编辑 backend/src/main/resources/application.yml 中 multi-agent 段：

  multi-agent:
    enabled: true              # 开关
    streaming:
      timeout-ms: 120000       # SSE 超时
    agents:
      weather-analysis:
        timeout-ms: 5000       # 单个智能体超时
    debate:
      enabled: true            # 辩论开关
      max-rounds: 2            # 最大辩论轮次
      consensus-threshold: 0.5 # 共识阈值

  常见问题排查

  # 后端 unhealthy → 查日志找原因
  docker logs tourism-backend --tail=50

  # Milvus 没起来 → 检查依赖服务
  docker compose ps   # 确认 etcd / minio 是 healthy

  # 多智能体不工作 → 检查智能体注册
  curl http://localhost:8080/api/multi-agent/status
  # 看 agentsRegistered 是否为 8

  # 智能体全走 Mock → 配置高德 API Key
  # 在 .env 中设置 MAP_API_KEY=你的高德key

  # 前端改动没生效 → 强制刷新
  # 浏览器按 Cmd+Shift+R（Mac）/ Ctrl+Shift+R（Windows）

  # 端口被占用 → 修改 .env 里的端口
  BACKEND_PORT=8081
  FRONTEND_PORT=3001
